cmake_minimum_required(VERSION 3.10)
project(BallotChain)

# Use C++17
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Set output directories for binaries
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

# Include directories
include_directories(
    ./blockchain
    ./crypto
    ./utils
)
include_directories(
    ${PROJECT_SOURCE_DIR}/libs/asio/asio/include
)

# Source files
file(GLOB_RECURSE BLOCKCHAIN_SRC ./blockchain/*.cpp)
file(GLOB_RECURSE CRYPTO_SRC ./crypto/*.cpp)
file(GLOB_RECURSE UTILS_SRC ./utils/*.cpp)

# Client executable
add_executable(client
    ./client/client.cpp
    ${BLOCKCHAIN_SRC}
    ${CRYPTO_SRC}
    ${UTILS_SRC}
)

# Server executable
add_executable(server
    ./server/server.cpp
    ${BLOCKCHAIN_SRC}
    ${CRYPTO_SRC}
    ${UTILS_SRC}
)
